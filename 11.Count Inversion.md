class Solution {
    static int inv;
    // Function to merge two halves and count inversions.
    static int merge(int[] arr, int l, int mid, int r) {
        int[] L = new int[mid - l + 1];
        int[] R = new int[r - mid];
        for (int i = 0; i < L.length; i++) {
            L[i] = arr[l + i];
        }
        for (int i = 0; i < R.length; i++) {
            R[i] = arr[mid + 1 + i];
        }

        int i = 0, j = 0, k = l, inversions = 0;
        while (i < L.length && j < R.length) {
            if (L[i] <= R[j]) {
                arr[k++] = L[i++];
            } else {
                arr[k++] = R[j++];
                // All remaining elements in L[] are greater than R[j]
                inv+=(L.length - i);
                inversions += (L.length - i);
            }
        }

        // Copy the remaining elements of L[], if any
        while (i < L.length) {
            arr[k++] = L[i++];
        }

        // Copy the remaining elements of R[], if any
        while (j < R.length) {
            arr[k++] = R[j++];
        }

        return inversions;
    }

    // Function to perform merge sort and count inversions.
    static int mergeSort(int[] arr, int l, int r) {
        int inversions = 0;
        if (l < r) {
            int mid = (l + r) / 2;

            // Count inversions in the left half
            inversions += mergeSort(arr, l, mid);

            // Count inversions in the right half
            inversions += mergeSort(arr, mid + 1, r);

            // Count split inversions while merging
            inversions += merge(arr, l, mid, r);
        }
        return inversions;
    }

    // Function to count inversions in the array.
    static int inversionCount(int[] arr) {
        inv=0;
        mergeSort(arr, 0, arr.length - 1);
        return inv;
    }
}
